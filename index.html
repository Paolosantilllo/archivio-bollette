alert("app caricata"); let data = JSON.parse(localStorage.getItem("archivio")) || []; let currentPath = []; const list = document.getElementById("folders"); const addBtn = document.getElementById("addFolder"); const backBtn = document.getElementById("backBtn"); const pathBox = document.getElementById("path"); const searchInput = document.getElementById("search"); function save() { localStorage.setItem("archivio", JSON.stringify(data)); } function getCurrentLevel() { let level = data; for (let i = 0; i < currentPath.length; i++) { level = level[currentPath[i]].sub; } return level; } function getPathNames() { let names = ["Home"]; let level = data; for (let i = 0; i < currentPath.length; i++) { let folder = level[currentPath[i]]; names.push(folder.name); level = folder.sub; } return names.join(" / "); } function render() { backBtn.style.display = currentPath.length === 0 ?
